SELECT a.airport_code, COUNT(b.boarding_no) as num_passen
FROM public.airports_data a 
JOIN public.FLIGHTS f ON a.airport_code = f.arrival_airport 
JOIN public.ticket_flights tf ON tf.flight_id = f.flight_id 
JOIN public.boarding_passes b ON b.ticket_no = tf.ticket_no

GROUP BY a.airport_code
ORDER BY num_passen ASC;
